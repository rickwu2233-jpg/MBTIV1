            const messageBox = document.createElement('div');

            messageBox.className = `fixed bottom-4 left-1/2 transform -translate-x-1/2 p-3 px-6 text-white rounded-lg shadow-xl z-50 opacity-100 transition-opacity duration-300 ${colorClass}`;

            container.appendChild(messageBox);

            // 3秒后开始淡出

            setTimeout(() => {

                messageBox.style.opacity = '0';

                messageBox.addEventListener('transitionend', () => messageBox.remove());

            }, 3000);

        }

        /**

         * 计算最终的人格类型

        function calculateResult() {

            // 计算每个维度的倾向

            const EI_COUNT = quizData.filter(q => q.dimension === 'EI').length;

            const SN_COUNT = quizData.filter(q => q.dimension === 'SN').length;

            const TF_COUNT = quizData.filter(q => q.dimension === 'TF').length;

            const JP_COUNT = quizData.filter(q => q.dimension === 'JP').length;


            // 逻辑: 如果得分大于总题数的一半，则倾向于'E', 'S', 'T', 'J' (值=1)

            const E_I = (scores.EI > EI_COUNT / 2) ? 'E' : 'I';

            const S_N = (scores.SN > SN_COUNT / 2) ? 'S' : 'N';

            const T_F = (scores.TF > TF_COUNT / 2) ? 'T' : 'F';

            const J_P = (scores.JP > JP_COUNT / 2) ? 'J' : 'P';


            finalResultType = `${E_I}${S_N}${T_F}${J_P}`;

            renderBriefResult();

        }
